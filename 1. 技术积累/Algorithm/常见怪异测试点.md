9. ```c++
   // Look and say sequence  1140 PAT甲级
   
   #include <iostream>
   using namespace std;
   int main(){
       int n, j;
       string d;
       cin >> d >> n;
       for(int cnt = 1; cnt < n; cnt ++){
           string res;
           for(int i = 0; i < d.length(); i = j){
               for(j = i; j < d.length() && d[j] == d[i]; j++);
               res += d[i] + to_string(j - i);
           }
           d = res;
       }
       cout << d << endl;
       return 0;
   }
   ```
